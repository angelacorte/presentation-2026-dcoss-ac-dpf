{{- $owner := .Get "owner" | default "angelacorte" -}} {{- $repo := .Get "repo" | default "Personalized-Template-Hugo-Reveal-Slides" -}} {{- $path := .Get "path" | default "LICENSE" -}} {{- $branch := .Get "branch" | default "master" -}} {{- $url := printf "https://api.github.com/repos/%s/%s/contents/%s?ref=%s" $owner $repo $path $branch -}} {{- $headers := dict "Accept" "application/vnd.github+json" -}} {{- with os.Getenv "GITHUB_TOKEN" -}} {{- if gt (len .) 1 -}} {{- $headers = merge $headers (dict "Authorization" .) -}} {{- end -}} {{- end -}} {{- $json := dict -}} {{- with try (resources.GetRemote $url (dict "headers" $headers)) -}} {{- with .Err -}} {{- errorf "%s" . -}} {{- else with .Value -}} {{- $json = .Content | transform.Unmarshal (dict "format" "json") -}} {{- else -}} {{- errorf "Unable to get remote resource %q" $url -}} {{- end -}} {{- end -}} {{- $content := $json.content | base64Decode | htmlUnescape -}} {{- $content := split $content "\n" -}} {{- $start := .Get "from" | default "1" | int | add -1 }} {{- $end := .Get "to" | default (len $content) | int }} {{- $end := sub $end $start }} {{- $content := after $start (first (add $start $end) $content) -}} {{- $content := delimit $content "\n" -}} {{- $language := .Get "language" | default ( cond (or (strings.HasSuffix $path ".kt") (strings.HasSuffix $path ".kts")) "kotlin" ( cond (strings.HasSuffix $path ".yml") "yaml" ( cond (strings.HasSuffix $path ".md") "markdown" ( replaceRE "^.*\\.(\\w+)$" "$1" $path ) ) ) ) -}} {{- highlight ($content) ($language) "" -}}
